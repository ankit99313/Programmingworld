#include <stdio.h>
int main()
{
int avg=0;
int sum=0;
int i=0;
int num[5];
printf("Enter the number:\n");
for(i=0; i<5; i++)
{
scanf("%d", &num[i]);
}
for(i=0; i<5; i++)
{
sum=sum+num[i];
}
avg=sum/4;
printf("Average of entered number is: %d", avg);
return 0;
}
