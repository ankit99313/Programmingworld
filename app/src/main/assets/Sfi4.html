#include <stdio.h>
int main()
{
FILE *fp;
int lines=0;
char filename[100];
char c;
printf("Enter file name: ");
scanf("%s", filename);
fp=fopen(filename, "r");
if(fp==NULL)
{
printf("could not open file %s", filename);
return 0;
}
for(c=getc(fp); c!=EOF; c=getc(fp))
if(c=='\n')
lines=lines+1;
fclose(fp);
printf("The file %s has %d lines\n", filename, lines);
return 0;
}
