include <stdio.h>
void main() {
int n1, n2, n3;
printf("Enter three integer numbers: \n");
scanf("%d %d %d", &n1, &n2, &n3);
if (n1 >= n2 && n1 >= n3)
printf("%d is the largest number.", n1);
else if (n2 >= n1 && n2 >= n3)
printf("%d is the largest number.", n2);
else
printf("%d is the largest number.", n3);
}

Enter three numbers:
122
234
421
421 is the largest number.
