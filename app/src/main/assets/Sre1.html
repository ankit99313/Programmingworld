#include <stdio.h>
int factorial(int n);
int main() {
int n;
printf("Enter a positive integer: ");
scanf("%d",&n);
int z=factorial(n);
printf("Factorial of %d = %d", n,z);
return 0;
}
int factorial(int n) {
if (n>=1)
return n*factorial(n-1);
else
return 1;
}

Enter a positive number: 6
Factorial of 6=720
