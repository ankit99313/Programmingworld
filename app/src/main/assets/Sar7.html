#include <stdio.h>
int main()
{
int a[100], i, j, Size, store;
printf("Please Enter the size of an array: ");
scanf("%d", &Size);
for (i = 0; i < Size; i++)
{
scanf("%d", &a[i]);
}
j = i - 1;
i = 0;
while (i < j)
{
store = a[i];
a[i] = a[j];
a[j] = store;
i++;
j--;
}
printf("Result of an Reverse array is: ");
for (i = 0; i < Size; i++)
{
printf("%d \t", a[i]);
}
return 0;
}
