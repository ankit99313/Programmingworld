1.Program to print solid rectangular star pattern

#include <stdio.h>
void rectangle(int n, int m)
{
int i, j;
for (i = 1; i <= n; i++)
{
for (j = 1; j <= m; j++)
{
printf("* ");
}
printf("\n");
}
}
int main()
{
int rows, columns;
printf("Enter the number of rows: ");
scanf("%d", &rows);
printf("Enter the number of columns: ");
scanf("%d", &columns);
printf("\n");
rectangle(rows, columns);
return 0;
}

Output:-
Enter the number of rows: 5
Enter the number of columns: 5
* * * * *
* * * * *
* * * * *
* * * * *
* * * * *
