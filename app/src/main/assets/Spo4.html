#include <stdio.h>
void exchange(int *, int *);
main()
{
int m, n;
m=100;
n=200;
printf("before exchange :m=%d, n=%d\n", m, n);
exchange(&m, &n);
printf("after exchange :m=%d, n=%d", m, n);
}
exchange(int *a, int *b)
{
int k;
k=*a;
*a=*b;
*b=k;
}

Output:
before exchange :m=100, n=200
after exchange :m=200,n=100
