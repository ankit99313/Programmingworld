#include <stdio.h>
void main()
{
int Number, Reminder, Sum=0;
printf("Please enter any number\n");
scanf("%d\n", &Number);
while(Number > 0)
{
Reminder = Number % 10;
Sum = Sum+ Reminder;
Number = Number / 10;
}
printf("Sum of the digits of given Number = %d", Sum);
}
