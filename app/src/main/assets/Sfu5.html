#include <stdio.h>
#include <stdbool.h>
bool checkYear(int year)
{
// If a year is multiple of 400,
// then it is a leap year
if (year % 400 == 0)
return true;
if (year % 100 == 0)
return false;
// Else If a year is muliplt of 4,
// then it is a leap year
if (year % 4 == 0)
return true;
return false;
}
int main()
{
int year;
printf("Enter the year: ");
scanf("%d",&year);
checkYear(year)? printf("Leap year"):
printf("Not a leap year");
return 0;
}
