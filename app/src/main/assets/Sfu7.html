#include <stdio.h>
void check(char ch)
{
if (ch >= 'A' && ch <= 'Z')
printf("%c is an upperCase character", ch);
else if (ch >= 'a' && ch <= 'z')
printf("%c is a lowerCase character", ch);
else
printf("%c is not an aplhabetic character", ch);
}
int main()
{
char ch;
printf("Enter a character: ");
scanf("%c", &ch);
check(ch);
return 0;
}
