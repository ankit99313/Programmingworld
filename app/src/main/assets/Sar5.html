#include <stdio.h>
int main()
{
int ar[500], size, i;
printf("Enter the size of array: ");
scanf("%d",&size);
printf("Enter the elements in array:\n");
for(i=0; i<size; i++)
{
scanf("%d", &ar[i]);
}
printf("Array in reverse order: ");
for(i=size-1; i>0; i--)
{
printf("%d\t", ar[i]);
}
return 0;
}
